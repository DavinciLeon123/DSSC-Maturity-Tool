        >
          {/* Selection indicator */}
          <span style={{
            width: '20px',
            height: '20px',
            borderRadius: '50%',
            border: `2px solid ${value === opt.value ? '#399e5a' : 'rgba(6,0,79,0.25)'}`,

            display: 'inline-flex',
            alignItems: 'center',

            {value === opt.value && (
              <span style={{ width: '8px', height: '8px', borderRadius: '50%', background: 'white', display: 'block' }} />

          </span>
          {opt.label}
        </button>
      ))}
    </div>
  );
}
