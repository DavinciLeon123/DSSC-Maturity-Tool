    ```python
    from typing import Optional
    from datetime import datetime
    from sqlmodel import SQLModel, Field, Column
    from sqlalchemy import Text

    class ComplianceReport(SQLModel, table=True):
        __tablename__ = "compliance_report"

        id: Optional[int] = Field(default=None, primary_key=True)
        initiative_id: int = Field(foreign_key="initiative.id", index=True, unique=True)
        html_content: str = Field(sa_column=Column(Text))
        generated_at: datetime = Field(default_factory=datetime.utcnow)
        questionnaire_version: str
        total_answers: int
        critical_count: int
        non_critical_count: int
        compliant_count: int
    ```

    Important: Use `sa_column=Column(Text)` for html_content — SQLModel maps `str` to VARCHAR by default, which is too small for full HTML reports (see research Pitfall 7).

    Important: `unique=True` on initiative_id means one active report per initiative. Regeneration MUST use `on_conflict_do_update` (not `session.add()`) to avoid IntegrityError (research Pitfall 4).

    **Step 4: Create backend/app/services/report_generator.py**

    ```python
    from pathlib import Path
    from jinja2 import Environment, FileSystemLoader, select_autoescape
    from datetime import datetime

    TEMPLATES_DIR = Path(__file__).parent.parent / "templates"

    def _get_jinja_env() -> Environment:
        return Environment(
            loader=FileSystemLoader(str(TEMPLATES_DIR)),
            autoescape=select_autoescape(["html"]),
        )

    def generate_html_report(
        initiative: dict,
        answers: list[dict],
        findings: list[dict],
        evidence_by_code: dict,
        mami_config: dict,
    ) -> str:
        env = _get_jinja_env()
        template = env.get_template("report.html")
        matrix = _build_matrix(answers, findings, mami_config)
        findings_detail = _build_findings_detail(answers, findings, evidence_by_code, mami_config)
        context = {
            "initiative": initiative,
            "generated_at": datetime.utcnow().strftime("%Y-%m-%d %H:%M UTC"),
            "matrix": matrix,
            "findings_detail": findings_detail,
            "critical_count": sum(1 for f in findings if f.get("severity") == "CRITICAL"),
            "non_critical_count": sum(1 for f in findings if f.get("severity") == "NON_CRITICAL"),
            "total_answers": len(answers),
            "mami_config": mami_config,
        }
        return template.render(**context)

    def _build_matrix(answers, findings, mami_config):
        finding_lookup = {f["mami_code"]: f for f in findings}
        answer_lookup = {a["mami_code"]: a for a in answers}
        categories = ["scheme", "participants", "data", "services"]
        dimensions = ["human_readable", "machine_readable", "trust_anchors"]
        matrix = {cat: {dim: {} for dim in dimensions} for cat in categories}
        for code in mami_config.get("codes", []):
            code_id = code["id"]
            cat = code["category"]
            dim = code["dimension"]
            if code_id in finding_lookup:
                f = finding_lookup[code_id]
                status = f.get("severity", "COMPLIANT") if f.get("status") == "FINDING" else f.get("status", "COMPLIANT")
            elif code_id in answer_lookup:
                a = answer_lookup[code_id]
                status = "NOT_APPLICABLE" if a["answer_value"] == "NOT_APPLICABLE" else "COMPLIANT"
            else:
                status = "UNANSWERED"
            matrix[cat][dim][code_id] = status
        return matrix

    def _build_findings_detail(answers, findings, evidence_by_code, mami_config):
        code_lookup = {c["id"]: c for c in mami_config.get("codes", [])}
        answer_lookup = {a["mami_code"]: a for a in answers}
        detail = []
        for f in findings:
            if f.get("status") != "FINDING":
                continue
            code = code_lookup.get(f["mami_code"], {})
            answer = answer_lookup.get(f["mami_code"], {})
            evidence = evidence_by_code.get(f["mami_code"], [])
            detail.append({
                "mami_code": f["mami_code"],
                "severity": f.get("severity", ""),
                "description": code.get("description", ""),
                "moscow_level": code.get("moscow_level", ""),
                "answer_value": answer.get("answer_value", ""),
                "rationale": answer.get("rationale", ""),
                "evidence": [
                    {"url": ev.url if hasattr(ev, "url") else ev.get("url", "")}
                    for ev in evidence
                ],
                "next_steps": _suggest_next_steps(f.get("severity", ""), code),
            })
        return detail

    def _suggest_next_steps(severity: str, code: dict) -> str:
        if severity == "CRITICAL":
            return f"This is a MUST requirement. Address {code.get('description', 'this requirement')} to achieve compliance."
        return f"This is a recommended improvement. Consider addressing {code.get('description', 'this item')}."
    ```

    **Step 5: Create backend/app/api/v1/reports.py** — POST generates report (scores answers, loads evidence, renders HTML, upserts to DB). GET returns stored HTML.

    ```python
    from fastapi import APIRouter, Depends, HTTPException
    from fastapi.responses import HTMLResponse
    from sqlmodel import Session, select
    from sqlalchemy.dialects.postgresql import insert as pg_insert
    from datetime import datetime
    import zen
    from app.core.deps import get_current_user, get_zen_engine, get_mami_config
    from app.db.session import get_session
    from app.models.user import User
    from app.models.initiative import Initiative
    from app.models.report import ComplianceReport
    from app.models.questionnaire import QuestionnaireAnswer
    from app.models.evidence import EvidenceURL
    from app.services.scoring_engine import score_all_answers
    from app.services.report_generator import generate_html_report

    router = APIRouter(tags=["reports"])

    @router.post("/initiatives/{initiative_id}/report", response_class=HTMLResponse)
    async def generate_report(
        initiative_id: int,
        session: Session = Depends(get_session),
        current_user: User = Depends(get_current_user),
        engine: zen.ZenEngine = Depends(get_zen_engine),
        mami_config: dict = Depends(get_mami_config),
    ):
        initiative = session.get(Initiative, initiative_id)
        if not initiative or initiative.user_id != current_user.id:
            raise HTTPException(status_code=404, detail="Initiative not found")

        answers = session.exec(
            select(QuestionnaireAnswer).where(QuestionnaireAnswer.initiative_id == initiative_id)
        ).all()

        code_lookup = {c["id"]: c for c in mami_config.get("codes", [])}
        answers_for_scoring = [
            {
                "mami_code": a.mami_code,
                "moscow_level": code_lookup.get(a.mami_code, {}).get("moscow_level", "SHOULD"),
                "answer_value": a.answer_value,
                "critical_override": code_lookup.get(a.mami_code, {}).get("critical_override"),
            }
            for a in answers
        ]
        findings_raw = await score_all_answers(engine, answers_for_scoring)

        evidence_rows = session.exec(
            select(EvidenceURL).where(EvidenceURL.initiative_id == initiative_id)
        ).all()
        evidence_by_code = {}
        for ev in evidence_rows:
            evidence_by_code.setdefault(ev.mami_code, []).append(ev)

        answers_dict = [
            {"mami_code": a.mami_code, "answer_value": a.answer_value, "rationale": a.rationale}
            for a in answers
        ]
        initiative_dict = {
            "name": initiative.name,
            "organization": initiative.organization,
            "contact_name": initiative.contact_name,
        }

        html_content = generate_html_report(
            initiative=initiative_dict,
            answers=answers_dict,
            findings=findings_raw,
            evidence_by_code=evidence_by_code,
            mami_config=mami_config,
        )

        # Upsert — one report per initiative, regeneration replaces
        stmt = pg_insert(ComplianceReport).values(
            initiative_id=initiative_id,
            html_content=html_content,
            generated_at=datetime.utcnow(),
            questionnaire_version="1.0",
            total_answers=len(answers),
            critical_count=sum(1 for f in findings_raw if f.get("severity") == "CRITICAL"),
            non_critical_count=sum(1 for f in findings_raw if f.get("severity") == "NON_CRITICAL"),
            compliant_count=len(answers) - sum(1 for f in findings_raw if f.get("status") == "FINDING"),
        ).on_conflict_do_update(
            index_elements=["initiative_id"],
            set_={
                "html_content": pg_insert(ComplianceReport).excluded.html_content,
                "generated_at": pg_insert(ComplianceReport).excluded.generated_at,
                "questionnaire_version": pg_insert(ComplianceReport).excluded.questionnaire_version,
                "total_answers": pg_insert(ComplianceReport).excluded.total_answers,
                "critical_count": pg_insert(ComplianceReport).excluded.critical_count,
                "non_critical_count": pg_insert(ComplianceReport).excluded.non_critical_count,
                "compliant_count": pg_insert(ComplianceReport).excluded.compliant_count,
            },
        )
        session.exec(stmt)
        session.commit()

        return HTMLResponse(content=html_content, status_code=200)

    @router.get("/initiatives/{initiative_id}/report", response_class=HTMLResponse)
    def get_report(
        initiative_id: int,
        session: Session = Depends(get_session),
        current_user: User = Depends(get_current_user),
    ):
        initiative = session.get(Initiative, initiative_id)
        if not initiative or initiative.user_id != current_user.id:
            raise HTTPException(status_code=404, detail="Initiative not found")
        report = session.exec(
            select(ComplianceReport).where(ComplianceReport.initiative_id == initiative_id)
        ).first()
        if not report:
            raise HTTPException(status_code=404, detail="No report generated yet")
        return HTMLResponse(content=report.html_content, status_code=200)
    ```

    ```typescript
    export async function generateReport(initiativeId: number): Promise<string> {
      const res = await api.post(`/initiatives/${initiativeId}/report`, null, {

        headers: { Accept: "text/html" },
    ```

    The `handleGenerateReport` function:
    ```typescript
    async function handleGenerateReport() {
      if (!initiativeId) return;
      setIsGenerating(true);
      try {
        await generateReport(initiativeId);
        // Open the stored report in a new tab
        const token = localStorage.getItem("token");
        const reportUrl = getReportUrl(initiativeId);
        window.open(reportUrl, "_blank");
      } finally {
        setIsGenerating(false);
      }
    }
    ```

    Note: The GET endpoint requires authentication. Since we use Bearer token auth, opening in a new tab won't automatically include the token. Two approaches:
    - **Option A (simplest for MVP):** The `generateReport()` POST returns the HTML as text. Create a blob URL from it and open that: `const blob = new Blob([html], { type: "text/html" }); window.open(URL.createObjectURL(blob), "_blank");`
    - **Option B:** Add a query parameter token option to the GET endpoint.

    Use Option A — it requires no backend changes and works immediately.

    ```typescript
    async function handleGenerateReport() {
      if (!initiativeId) return;
      setIsGenerating(true);
      try {
        const html = await generateReport(initiativeId);
        const blob = new Blob([html], { type: "text/html" });
        window.open(URL.createObjectURL(blob), "_blank");
      } finally {
        setIsGenerating(false);
      }
    }
    ```
