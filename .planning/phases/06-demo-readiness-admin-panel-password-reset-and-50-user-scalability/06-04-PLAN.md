```typescript
type SaveBadgeState = "idle" | "saving" | "saved" | "failed" | "rate-limited";

interface AutosaveBadgeProps {
  state: SaveBadgeState;
}

function AutosaveBadge({ state }: AutosaveBadgeProps) {
  if (state === "saving") {
    return (
      <span style={{ fontSize: "0.75rem", color: "#6B7280" }}>
        Saving...
      </span>
    );
  }
  if (state === "saved") {
    return (
      <span style={{ fontSize: "0.75rem", color: "#059669", fontWeight: 500 }}>
        Saved
      </span>
    );
  }
  if (state === "failed") {
    return (
      <span style={{ fontSize: "0.75rem", color: "#F59E0B", fontWeight: 500 }}>
        Save failed — retrying
      </span>
    );
  }
  if (state === "rate-limited") {
    return (
      <span style={{ fontSize: "0.75rem", color: "#F59E0B", fontWeight: 500 }}>
        Too many saves — slow down
      </span>
    );
  }
  return null;
}
```

**Change 3 — Wire badge state to save mutation and implement 429 auto-retry:**

The existing `saveMutation` must be updated to drive badgeState instead of (or in addition to) its current callbacks. Modify the mutation to add explicit badge state management:

```typescript
const saveMutation = useMutation({
  mutationFn: async (payload: /* existing type */) => {
    setBadgeState("saving");
    return saveAnswer(/* existing args */);  // keep existing function call
  },
  onSuccess: () => {
    setBadgeState("saved");
    // Auto-clear "Saved" back to idle after 2 seconds
    setTimeout(() => setBadgeState("idle"), 2000);
  },
  onError: (error: unknown) => {
    const status = (error as { response?: { status?: number } }).response?.status;
    if (status === 429) {
      setBadgeState("rate-limited");
      // Auto-retry after 3 seconds (per CONTEXT.md locked decision)
      setTimeout(() => {
        setBadgeState("idle");
        // Note: auto-retry triggers when user next saves; we don't replay the last payload here
        // The badge returns to idle and the user can trigger the next save normally
      }, 3000);
    } else {
      setBadgeState("failed");
    }
  },
});
```
