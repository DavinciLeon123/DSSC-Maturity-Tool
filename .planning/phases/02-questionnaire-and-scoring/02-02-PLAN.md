    ```python
    from enum import Enum
    from typing import Optional
    from datetime import datetime
    from sqlmodel import SQLModel, Field
    from sqlalchemy import UniqueConstraint

    class AnswerValue(str, Enum):
        yes = "YES"
        no = "NO"
        comply_explain = "COMPLY_EXPLAIN"
        not_applicable = "NOT_APPLICABLE"

    class QuestionnaireAnswer(SQLModel, table=True):
        __tablename__ = "questionnaire_answer"
        __table_args__ = (
            UniqueConstraint("initiative_id", "question_id", name="uq_answer_per_question"),
        )

        id: Optional[int] = Field(default=None, primary_key=True)
        initiative_id: int = Field(foreign_key="initiative.id", index=True)
        question_id: str = Field(index=True)          # e.g. "q_S-HRA-1.1"
        mami_code: str = Field(index=True)             # e.g. "S-HRA-1.1" (denormalized)
        questionnaire_version: str                     # e.g. "1.0" — QUES-07
        answer_value: AnswerValue
        rationale: Optional[str] = None               # Required only for COMPLY_EXPLAIN
        answered_at: datetime = Field(default_factory=datetime.utcnow)
        updated_at: datetime = Field(default_factory=datetime.utcnow)
    ```

    ```python
    from typing import Optional
    from pydantic import BaseModel, model_validator
    from app.models.questionnaire import AnswerValue

    class AnswerCreate(BaseModel):
        question_id: str
        mami_code: str
        questionnaire_version: str
        answer_value: AnswerValue
        rationale: Optional[str] = None

        @model_validator(mode="after")
        def rationale_required_for_comply_explain(self):
            if self.answer_value == AnswerValue.comply_explain and not self.rationale:
                raise ValueError("rationale is required when answer_value is COMPLY_EXPLAIN")
            return self

    class AnswerRead(BaseModel):
        id: int
        initiative_id: int
        question_id: str
        mami_code: str
        questionnaire_version: str
        answer_value: AnswerValue

            from_attributes = True
    ```

    **Step 3: Create backend/app/api/v1/questionnaire.py**

    Two endpoints:
    - `GET /questionnaire/config` — returns the questionnaire config from app.state (no auth needed for browsing, but add JWT auth for consistency with the app)
    - `PUT /initiatives/{initiative_id}/answers/{question_id}` — upsert one answer using PostgreSQL on_conflict_do_update

    ```python
    from fastapi import APIRouter, Depends, HTTPException
    from sqlmodel import Session, select
    from sqlalchemy.dialects.postgresql import insert as pg_insert
    from datetime import datetime
    from app.db.session import get_session
    from app.core.deps import get_current_user, get_questionnaire_config
    from app.models.user import User
    from app.models.initiative import Initiative
    from app.models.questionnaire import QuestionnaireAnswer
    from app.schemas.questionnaire import AnswerCreate, AnswerRead

    router = APIRouter(prefix="/questionnaire", tags=["questionnaire"])

    @router.get("/config")
    def get_questionnaire_config_endpoint(
        config: dict = Depends(get_questionnaire_config),
        current_user: User = Depends(get_current_user),
    ):
        """Return the questionnaire config from app.state (loaded from questionnaire-v1.json)."""
        return config

    @router.put("/initiatives/{initiative_id}/answers/{question_id}", response_model=AnswerRead)
    def upsert_answer(
        initiative_id: int,
        question_id: str,
        answer_in: AnswerCreate,
        session: Session = Depends(get_session),
        current_user: User = Depends(get_current_user),
    ):
        """Upsert one answer for a question. Creates on first save, updates on subsequent saves.
        Enforces ownership: current user must own the initiative."""
        # Verify initiative ownership
        initiative = session.get(Initiative, initiative_id)
        if not initiative:
            raise HTTPException(status_code=404, detail="Initiative not found")
        if initiative.user_id != current_user.id:
            raise HTTPException(status_code=403, detail="Not your initiative")

        # PostgreSQL upsert (insert or update on conflict)
        stmt = pg_insert(QuestionnaireAnswer).values(
            initiative_id=initiative_id,
            question_id=question_id,
            mami_code=answer_in.mami_code,
            questionnaire_version=answer_in.questionnaire_version,
            answer_value=answer_in.answer_value,
            rationale=answer_in.rationale,
            answered_at=datetime.utcnow(),
            updated_at=datetime.utcnow(),
        )
        stmt = stmt.on_conflict_do_update(
            constraint="uq_answer_per_question",
            set_={
                "answer_value": stmt.excluded.answer_value,
                "rationale": stmt.excluded.rationale,
                "questionnaire_version": stmt.excluded.questionnaire_version,
                "updated_at": stmt.excluded.updated_at,
            },
        )
        session.exec(stmt)
        session.commit()

        # Fetch and return the upserted row
        result = session.exec(
            select(QuestionnaireAnswer).where(
                QuestionnaireAnswer.initiative_id == initiative_id,
                QuestionnaireAnswer.question_id == question_id,
            )
        ).one()
        return result

    @router.get("/initiatives/{initiative_id}/answers", response_model=list[AnswerRead])
    def get_answers(
        initiative_id: int,
        session: Session = Depends(get_session),
        current_user: User = Depends(get_current_user),
    ):
        """Return all saved answers for an initiative (for save/resume — QUES-02)."""
        initiative = session.get(Initiative, initiative_id)
        if not initiative:
            raise HTTPException(status_code=404, detail="Initiative not found")
        if initiative.user_id != current_user.id:
            raise HTTPException(status_code=403, detail="Not your initiative")

        answers = session.exec(
            select(QuestionnaireAnswer).where(
                QuestionnaireAnswer.initiative_id == initiative_id
            )
        ).all()
        return answers
    ```

    ```typescript
      dimensions: Dimension[];

    }

    export interface AnswerPayload {
      question_id: string;
      mami_code: string;
      questionnaire_version: string;
      answer_value: "YES" | "NO" | "COMPLY_EXPLAIN" | "NOT_APPLICABLE";
      rationale?: string;
    }

    export interface AnswerRecord {
      id: number;
      question_id: string;
      answer_value: string;
      rationale: string | null;
      questionnaire_version: string;
    }

    export async function fetchQuestionnaireConfig(): Promise<QuestionnaireConfig> {
      const res = await api.get<QuestionnaireConfig>("/questionnaire/config");
      return res.data;
    }

    export async function fetchAnswers(initiativeId: number): Promise<AnswerRecord[]> {
      const res = await api.get<AnswerRecord[]>(`/questionnaire/initiatives/${initiativeId}/answers`);

      initiativeId: number,
      question_id: string,
      payload: AnswerPayload
    ): Promise<AnswerRecord> {
      const res = await api.put<AnswerRecord>(
        `/questionnaire/initiatives/${initiativeId}/answers/${question_id}`,
        payload
      );
      return res.data;
    ```

    ```tsx
    import { WidgetProps } from "@rjsf/utils";

    export function NotApplicableWidget({ value, onChange, label, schema }: WidgetProps) {
      const options = ["YES", "NO", "COMPLY_EXPLAIN", "NOT_APPLICABLE"];
      // When value is a composite {value, rationale} object (from COMPLY_EXPLAIN), unwrap
      const currentAnswer = typeof value === "object" && value !== null ? value.value : value;
      const currentRationale = typeof value === "object" && value !== null ? value.rationale : "";

      return (
        <div className="question-widget">
          <div className="answer-options" role="radiogroup">
            {options.map((opt) => (
              <label key={opt} className="radio-option">
                <input
                  type="radio"
                  name={String(schema.title || label)}
                  value={opt}
                  checked={currentAnswer === opt}
                  onChange={() => {
                    if (opt === "COMPLY_EXPLAIN") {
                      onChange({ value: "COMPLY_EXPLAIN", rationale: currentRationale || "" });
                    } else {
                      onChange(opt);
                    }
                  }}
                />
                <span>{opt === "COMPLY_EXPLAIN" ? "Comply or Explain" : opt.replace(/_/g, " ")}</span>
              </label>
            ))}
          </div>
          {currentAnswer === "COMPLY_EXPLAIN" && (
            <textarea
              className="rationale-textarea"
              placeholder="Explain how you comply or why you cannot..."
              value={currentRationale}
              onChange={(e) =>
                onChange({ value: "COMPLY_EXPLAIN", rationale: e.target.value })
              }
            />
          )}
        </div>
      );
    }
    ```

    **Step 5: Create frontend/src/components/questionnaire/QuestionnaireForm.tsx**

    RJSF form wrapper. Builds a JSON Schema from the questions array for a single category/dimension. Uses custom widgets per question answer_type. The form's `onChange` handler calls `onAnswerChange` for each changed field:

    ```tsx
    import Form from "@rjsf/core";
    import validator from "@rjsf/validator-ajv8";
    import { RJSFSchema, UiSchema } from "@rjsf/utils";
    import { NotApplicableWidget } from "./NotApplicableWidget";
    import { ComplyExplainWidget } from "./ComplyExplainWidget";
    import type { Question } from "../../lib/questionnaire";

    const customWidgets = {
      notApplicableWidget: NotApplicableWidget,
      complyExplainWidget: ComplyExplainWidget,
    };

    interface Props {
      questions: Question[];
      formData: Record<string, unknown>;
      onAnswerChange: (questionId: string, value: unknown) => void;
    }

    function buildSchema(questions: Question[]): RJSFSchema {
      const properties: Record<string, RJSFSchema> = {};
      questions.forEach((q) => {
        properties[q.id] = {
          type: "string",
          title: q.text,
          enum: q.answer_type === "not_applicable_allowed"
            ? ["YES", "NO", "COMPLY_EXPLAIN", "NOT_APPLICABLE"]
            : ["YES", "NO", "COMPLY_EXPLAIN"],
        };
      });
      return {
        type: "object",
        properties,
        required: questions.filter((q) => q.required).map((q) => q.id),
      };
    }

    function buildUiSchema(questions: Question[]): UiSchema {
      const uiSchema: UiSchema = {};
      questions.forEach((q) => {
        if (q.answer_type === "not_applicable_allowed") {
          uiSchema[q.id] = { "ui:widget": "notApplicableWidget" };
        } else {
          uiSchema[q.id] = { "ui:widget": "complyExplainWidget" };
        }
      });
      return uiSchema;
    }

    export function QuestionnaireForm({ questions, formData, onAnswerChange }: Props) {
      const schema = buildSchema(questions);
      const uiSchema = buildUiSchema(questions);

      const handleChange = ({ formData: newData }: { formData: Record<string, unknown> }) => {
        Object.entries(newData ?? {}).forEach(([questionId, value]) => {
          if (value !== formData[questionId]) {
            onAnswerChange(questionId, value);
          }
        });
      };

      return (
        <Form
          schema={schema}
          uiSchema={uiSchema}
          formData={formData}
          validator={validator}
          widgets={customWidgets}
          onChange={handleChange}
        >
          {/* Suppress default submit button — saving is handled via onAnswerChange */}
          <div />
        </Form>
      );
    ```

    Main questionnaire page. Uses TanStack Query to load config and existing answers. Category tabs at top (4 tabs). Within each category, dimensions are sections. QuestionnaireForm renders questions per dimension. Auto-save on each answer change via mutation:

    ```tsx
    import { useState, useMemo } from "react";
    import { useQuery, useMutation } from "@tanstack/react-query";
    import {
      fetchQuestionnaireConfig,
      fetchAnswers,
      saveAnswer,
      type Category,
    } from "../../lib/questionnaire";
    import { QuestionnaireForm } from "../../components/questionnaire/QuestionnaireForm";
    import { createFileRoute } from "@tanstack/react-router";

    export const Route = createFileRoute("/_app/questionnaire")({
      component: QuestionnairePage,
    });

    function QuestionnairePage() {
      // Fetch user's initiative ID from the initiative endpoint (reuse existing logic)
      // For now, get initiativeId from /api/v1/initiatives/me
      const { data: initiative } = useQuery({
        queryKey: ["initiative"],
        queryFn: async () => {
          const api = (await import("../../lib/api")).default;
          const res = await api.get("/initiatives/me");
          return res.data as { id: number };
        },
      });

      const initiativeId = initiative?.id;

      const { data: config, isLoading: configLoading } = useQuery({
        queryKey: ["questionnaire-config"],
        queryFn: fetchQuestionnaireConfig,
      });

      const { data: savedAnswers = [] } = useQuery({
        queryKey: ["questionnaire-answers", initiativeId],
        queryFn: () => fetchAnswers(initiativeId!),
        enabled: !!initiativeId,
      });

      // Build formData map: { question_id: answer_value | {value, rationale} }
      const formData = useMemo(() => {
        const map: Record<string, unknown> = {};
        savedAnswers.forEach((a) => {
          if (a.answer_value === "COMPLY_EXPLAIN" && a.rationale) {
            map[a.question_id] = { value: "COMPLY_EXPLAIN", rationale: a.rationale };
          } else {
            map[a.question_id] = a.answer_value;
          }
        });
        return map;
      }, [savedAnswers]);

      const [activeCategory, setActiveCategory] = useState<string>("");

      const activeCategoryId =
        activeCategory || config?.categories?.[0]?.id || "";

      const saveMutation = useMutation({
        mutationFn: async ({ questionId, value, question }: {
          questionId: string;
          value: unknown;
          question: { mami_code: string };
        }) => {
          if (!initiativeId || !config) return;
          const answerValue = typeof value === "object" && value !== null
            ? (value as { value: string }).value
            : (value as string);
          const rationale = typeof value === "object" && value !== null
            ? (value as { rationale: string }).rationale
            : undefined;

          await saveAnswer(initiativeId, questionId, {
            question_id: questionId,
            mami_code: question.mami_code,
            questionnaire_version: config.version,
            answer_value: answerValue as "YES" | "NO" | "COMPLY_EXPLAIN" | "NOT_APPLICABLE",
            rationale,
          });
        },
      });

      if (configLoading || !config) {
        return <div className="page-loading">Loading questionnaire...</div>;
      }

      if (!initiativeId) {
        return (
          <div className="page-content">
            <p>Please create your DSI initiative first before filling in the questionnaire.</p>
          </div>
        );
      }

      const activeCategory_obj: Category | undefined = config.categories.find(
        (c) => c.id === activeCategoryId
      );

      // Flatten question lookup map for save mutation
      const questionMap = new Map(
        config.categories.flatMap((cat) =>
          cat.dimensions.flatMap((dim) =>
            dim.questions.map((q) => [q.id, q])
          )
        )
      );

      return (
        <div className="questionnaire-page">
          <h1>MAMI Questionnaire</h1>
          <p className="questionnaire-subtitle">
            Version {config.version} — Answers auto-save as you go
          </p>

          {/* Category tabs */}
          <div className="category-tabs" role="tablist">
            {config.categories.map((cat) => (
              <button
                key={cat.id}
                role="tab"
                aria-selected={cat.id === activeCategoryId}
                className={`category-tab ${cat.id === activeCategoryId ? "active" : ""}`}
                onClick={() => setActiveCategory(cat.id)}
              >
                {cat.label}
              </button>
            ))}
          </div>

          {/* Active category content */}
          {activeCategory_obj && (
            <div className="category-content">
              {activeCategory_obj.dimensions.map((dim) => (
                <section key={dim.id} className="dimension-section">
                  <h2 className="dimension-title">{dim.label}</h2>
                  <QuestionnaireForm
                    questions={dim.questions}
                    formData={formData}
                    onAnswerChange={(questionId, value) => {
                      const question = questionMap.get(questionId);
                      if (question) {
                        saveMutation.mutate({ questionId, value, question });
                      }
                    }}
                  />
                </section>
              ))}
            </div>
          )}

          {saveMutation.isPending && (
            <div className="save-indicator">Saving...</div>
          )}
        </div>
      );
    }
    ```
